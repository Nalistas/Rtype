\begin{DoxyItemize}
\item -\/o, --output \mbox{[}output file\mbox{]} Specify file output. If none is specified, write to stdout.
\item -\/i, --input \mbox{[}input file\mbox{]} Specify file input, otherwise use last argument as input file. If no input file is specified, read from stdin.
\item -\/s, --string \mbox{[}markdown string\mbox{]} Specify string input instead of a file.
\item -\/c, --config \mbox{[}config file\mbox{]} Specify config file to use instead of the default {\ttfamily \texorpdfstring{$\sim$}{\string~}/.marked.\+json} or {\ttfamily \texorpdfstring{$\sim$}{\string~}/.marked.\+js} or {\ttfamily \texorpdfstring{$\sim$}{\string~}/.marked/index.\+js}.
\item -\/t, --tokens Output a token list instead of html.
\item -\/n, --no-\/clobber Do not overwrite {\ttfamily output} if it exists.
\item --pedantic Conform to obscure parts of markdown.\+pl as much as possible. Don\textquotesingle{}t fix original markdown bugs.
\item --gfm Enable github flavored markdown.
\item --breaks Enable GFM line breaks. Only works with the gfm option.
\item --no-\/breaks, -\/no-\/etc... The inverse of any of the marked options above.
\item --silent Silence error output.
\item -\/h, --help Display help information.
\end{DoxyItemize}\hypertarget{md__2home_2solype_2delivery_2current__days_2Mannheim_2front_2node__modules_2marked_2man_2marked_81_autotoc_md3837}{}\doxysection{\texorpdfstring{CONFIGURATION}{CONFIGURATION}}\label{md__2home_2solype_2delivery_2current__days_2Mannheim_2front_2node__modules_2marked_2man_2marked_81_autotoc_md3837}

\begin{DoxyCode}{0}

But the function will iterate through each declaration\+:


\begin{DoxyCode}{0}
\DoxyCodeLine{parse('color:\ \#f00',\ (name,\ value,\ declaration)\ =>\ \{}
\DoxyCodeLine{\ \ console.log(name);\ \ \ \ \ \ \ \ //\ 'color'}
\DoxyCodeLine{\ \ console.log(value);\ \ \ \ \ \ \ //\ '\#f00'}
\DoxyCodeLine{\ \ console.log(declaration);\ //\ \{\ type:\ 'declaration',\ property:\ 'color',\ value:\ '\#f00'\ \}}
\DoxyCodeLine{\});}

\end{DoxyCode}
